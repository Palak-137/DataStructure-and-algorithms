import java.io.*;
import java.math.*;
import java.text.*;
import java.util.*;
import java.util.regex.*;

public class Solution {

    public static String timeconversion(String input,String output , String givendate )
    throws ParseException{
        SimpleDateFormat inputformat = new SimpleDateFormat(input);
        SimpleDateFormat outputformat = new SimpleDateFormat(output);
        Date date = null;
        String requireddate = null;
        date = inputformat.parse(givendate);
        requireddate = outputformat.format(date);
        return requireddate;
    }
   public static void main(String [] args)throws ParseException{
    Scanner scan = new Scanner(System.in);
    String date = scan.nextLine();
    String outputdate = timeconversion("hh:mm:ssa","HH:mm:ss",date);
     System.out.println(outputdate);
}
}
